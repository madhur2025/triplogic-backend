const mongoose = require('mongoose')

const connectDB = async () => {
    try {
        const conn = await mongoose.connect('mongodb://localhost:27017')
        console.log("Connected to mongoDB")
    }
    catch (err) {
        console.error(err.message)
        process.exit(1)
    }
}
module.exports = connectDB
